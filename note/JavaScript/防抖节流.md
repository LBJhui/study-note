# 防抖

1. 对于在事件被触发 n 秒后再执行的会调 ->延迟执行
2. 如果在这 n 秒内再触发事件，重新开始计时

```javascript
function debounce (fn, time, triggleNow) {
  var t = null
  var res

  var debounced = function () {
    var _self = this
    var args = arguments

    if (t) {
      clearTimeout(t)
    }

    if (triggleNow) {
      var exec = !t

      t = setTimeout(function () {
        t = null
      }, time)

      if (exec) {
        res = fn.apply(_self, args)
      }
    } else {
      t = setTimeout(function () {
        res = fn.apply(_self, args)
      }, time)
    }
    return res
  }
  debounced.remove = function () {
    clearTimeout(t)
    t = null
  }
},
```
